! Allowed-edge ACL policy
! Networks are interpreted as 192.168.X.0/24
! Any routed flow not listed below is blocked.

! =========================
! R0 (CSE: 10, 11, 12)
! =========================
interface g0/2.10
 no ip access-group R0 in
 no ip access-group R0 out
exit
interface g0/2.20
 no ip access-group R0 in
 no ip access-group R0 out
exit
interface g0/2.30

exit
no ip access-list extended R0
ip access-list extended R0
 permit udp any eq bootpc any eq bootps

 permit ip 192.168.10.0 0.0.0.255 192.168.40.0 0.0.0.255

 permit ip 192.168.12.0 0.0.0.255 host 192.168.50.6

exit

exit

! =========================
! R6 (CENTRAL INTER-INSTITUTION CONTROL)
! =========================
! R6 controls these flows so R7 does not need changes:
! (1) Block 192.168.70.0/24 from server 192.168.100.2
! (2) Allow traffic between 192.168.70.6 and 192.168.50.6
no ip access-list extended R6
ip access-list extended R6
 deny ip 192.168.70.0 0.0.0.255 host 192.168.100.2
 permit ip host 192.168.70.6 host 192.168.50.6
 permit ip host 192.168.50.6 host 192.168.70.6
 permit ip host 192.168.100.2 any
 permit ip any host 192.168.100.2
 permit ip any any
exit
